### 2. QAnything API集成（流式输出）
```typescript
// app/api/qanything/route.ts
import { NextResponse } from 'next/server';

export async function POST(request: Request) {
  const { question, kbId } = await request.json();
  const apiKey = process.env.QANYTHING_API_KEY;
  const appKey = process.env.QANYTHING_APP_KEY;
  
  const response = await fetch(`${process.env.QANYTHING_BASE_URL}/api/local_doc_qa/local_doc_chat`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${apiKey}`,
    },
    body: JSON.stringify({
      user_id: 'default',
      kb_ids: [kbId],
      question,
      streaming: true,
      rerank: true
    }),
  });
  
  const stream = new ReadableStream({
    async start(controller) {
      const reader = response.body?.getReader();
      if (!reader) return;
      
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        controller.enqueue(value);
      }
      controller.close();
    }
  });
  
  return new NextResponse(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
    },
  });
}
```

   export async function getFundData(fundCode: string) {
     const apiKey = process.env.TUSHARE_API_KEY;
     if (!apiKey) throw new Error("TUSHARE_API_KEY not set");
     
     const response = await fetch('http://api.tushare.pro', {
       method: 'POST',
       headers: { 'Content-Type': 'application/json' },
       body: JSON.stringify({
         api_name: 'funda_nav',
         token: apiKey,
         params: { ts_code: fundCode },
         fields: 'end_date,unit_nav,累计净值'
       })
     });
     
     const data = await response.json();
     if (data.code !== 0) throw new Error(data.msg);
     
     return data.data.items.map(item => ({
       date: item[0],
       unitNav: parseFloat(item[1]),
       totalNav: parseFloat(item[2])
     }));
   }

   // components/FundAgentChat.tsx
   import { useState } from 'react';
   import { motion } from 'framer-motion';
   
   export default function FundAgentChat() {
     const [messages, setMessages] = useState([]);
     const [input, setInput] = useState('');
     
     const handleSend = async () => {
       if (!input.trim()) return;
       
       const newMessages = [...messages, { role: 'user', content: input }];
       setMessages(newMessages);
       setInput('');
       
       // 调用QAnything Agent API获取回复
       const response = await fetch('/api/qanything/agent/chat', {
         method: 'POST',
         body: JSON.stringify({ message: input, agentId: 'your_agent_id' })
       });
       
       const data = await response.json();
       setMessages([...newMessages, { role: 'assistant', content: data.answer }]);
     };
     
     return (
       <div className="agent-chat">
         <div className="chat-messages">
           {messages.map((msg, i) => (
             <div key={i} className={`message ${msg.role}`}>
               <motion.div
                 initial={{ opacity: 0, y: 10 }}
                 animate={{ opacity: 1, y: 0 }}
               >
                 {msg.content}
               </motion.div>
             </div>
           ))}
         </div>
         <div className="chat-input">
           <input
             type="text"
             value={input}
             onChange={(e) => setInput(e.target.value)}
             onKeyPress={(e) => e.key === 'Enter' && handleSend()}
             placeholder="向基金智能助手提问..."
           />
           <button onClick={handleSend}>发送</button>
         </div>
       </div>
     );
   }

// lib/api/wakatime.ts
export async function getWakaTimeStats() {
  const apiKey = process.env.WAKATIME_API_KEY;
  const userId = "waka_431cb8ec-e76f-444d-805b-734e1d62b5c5";

// lib/utils/date.ts
export function formatDuration(hours: number): string {
  const totalMinutes = Math.round(hours * 60);
  const hoursPart = Math.floor(totalMinutes / 60);
  const minutesPart = totalMinutes % 60;
  
  if (hoursPart === 0) {
    return `${minutesPart}分钟`;
  }
  
  return `${hoursPart}小时${minutesPart > 0 ? `${minutesPart}分钟` : ''}`;
}

export default function AnimatedCounter({ value, suffix = '' }) {
  const count = useMotionValue(0);
  const rounded = useTransform(count, Math.round);
  
  useEffect(() => {
    const controls = animate(count, value, {
      duration: 2,
      ease: "easeOut"
    });
    
    return controls.stop;
  }, [value]);
  
  return (
    <motion.span className="counter">
      {rounded}
      {suffix}
    </motion.span>
  );
}

### 动画性能优化

1. **减少动画元素数量**：避免同时动画过多元素，使用交错动画分散性能压力
2. **使用will-change**：对频繁动画的元素添加will-change属性
3. **硬件加速**：优先使用transform和opacity属性进行动画
4. **懒加载动画**：使用viewport触发动画，避免初始加载时的性能压力
5. **动画分层**：使用z-index和position: fixed分离动画元素，减少重排范围

```css
/* styles/animation-optimization.css */
.animated-element {
  will-change: transform, opacity;
  transform: translateZ(0); /* 触发硬件加速 */
}

/* 减少重绘 */
.avoid-repaint {
  contain: layout paint;
}

```tsx
// components/games/CodeMemoryGame.tsx
import { useState, useEffect, useRef } from 'react';
import { motion } from 'framer-motion';

export default function CodeMemoryGame() {
  const [cards, setCards] = useState([]);
  const [flippedIndices, setFlippedIndices] = useState([]);
  const [matchedPairs, setMatchedPairs] = useState(0);
  const [gameStarted, setGameStarted] = useState(false);
  const [timeElapsed, setTimeElapsed] = useState(0);
  const timerRef = useRef(null);
  
  // 代码片段数据集
  const codeSnippets = [
    { id: 1, text: 'function component() {}', category: 'js' },
    { id: 2, text: 'const [state, setState] = useState()', category: 'react' },
    { id: 3, text: 'useEffect(() => {}, [])', category: 'react' },
    { id: 4, text: 'async function fetchData() {}', category: 'js' },
    { id: 5, text: 'const router = useRouter()', category: 'nextjs' },
    { id: 6, text: 'export default function Page() {}', category: 'nextjs' },
  ];
  
  // 初始化游戏卡片
  useEffect(() => {
    // 创建配对卡片并打乱顺序
    const pairedCards = [...codeSnippets, ...codeSnippets]
      .sort(() => Math.random() - 0.5)
      .map((snippet, index) => ({
        ...snippet,
        index,
        flipped: false,
        matched: false
      }));
    
    setCards(pairedCards);
  }, []);
  
  // 开始游戏计时
  useEffect(() => {
    if (gameStarted && matchedPairs < codeSnippets.length) {
      timerRef.current = setInterval(() => {
        setTimeElapsed(prev => prev + 1);
      }, 1000);
    } else if (matchedPairs === codeSnippets.length) {
      clearInterval(timerRef.current);
    }
    
    return () => clearInterval(timerRef.current);
  }, [gameStarted, matchedPairs]);
  
  // 处理卡片点击
  const handleCardClick = (index) => {
    if (!gameStarted) setGameStarted(true);
    
    // 忽略已翻开或已匹配的卡片
    if (flippedIndices.length >= 2 || cards[index].flipped || cards[index].matched) 
      return;
    
    const newFlippedIndices = [...flippedIndices, index];
    const newCards = [...cards];
    newCards[index].flipped = true;
    setCards(newCards);
    setFlippedIndices(newFlippedIndices);
    
    // 检查匹配
    if (newFlippedIndices.length === 2) {
      const [firstIndex, secondIndex] = newFlippedIndices;
      if (cards[firstIndex].id === cards[secondIndex].id) {
        // 匹配成功
        newCards[firstIndex].matched = true;
        newCards[secondIndex].matched = true;
        setCards(newCards);
        setMatchedPairs(prev => prev + 1);
      }
      
      // 延迟后翻回未匹配的卡片
      setTimeout(() => {
        if (cards[firstIndex].id !== cards[secondIndex].id) {
          newCards[firstIndex].flipped = false;
          newCards[secondIndex].flipped = false;
          setCards(newCards);
        }
        setFlippedIndices([]);
      }, 1000);
    }
  };
  
  // 重置游戏
  const resetGame = () => {
    const pairedCards = [...codeSnippets, ...codeSnippets]
      .sort(() => Math.random() - 0.5)
      .map((snippet, index) => ({
        ...snippet,
        index,
        flipped: false,
        matched: false
      }));
    
    setCards(pairedCards);
    setFlippedIndices([]);
    setMatchedPairs(0);
    setGameStarted(false);
    setTimeElapsed(0);
  };
  
  // 格式化时间
  const formatTime = (seconds) => {
    const mins = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
  };
  
  return (
    <div className="code-memory-game">
      <div className="game-header">
        <h2>代码记忆配对游戏</h2>
        <div className="game-stats">
          <span>时间: {formatTime(timeElapsed)}</span>
          <span>匹配: {matchedPairs}/{codeSnippets.length}</span>
          <button onClick={resetGame} className="reset-btn">重置</button>
        </div>
      </div>
      
      {matchedPairs === codeSnippets.length && gameStarted && (
        <div className="game-over">
          <h3>恭喜你完成了游戏！</h3>
          <p>用时: {formatTime(timeElapsed)}</p>
          <button onClick={resetGame}>再玩一次</button>
        </div>
      )}
      
      <div className="card-grid">
        {cards.map((card) => (
          <motion.div
            key={card.index}
            className={`card ${card.flipped || card.matched ? 'flipped' : ''}`}
            onClick={() => handleCardClick(card.index)}
            initial={{ opacity: 0, scale: 0.8 }}
            animate={{ opacity: 1, scale: 1 }}
            transition={{ duration: 0.3 }}
          >
            <div className="card-inner">
              <div className="card-front">
                <div className="code-icon">?</div>
              </div>
              <div className="card-back">
                <pre className={`language-${card.category}`}>
                  <code>{card.text}</code>
                </pre>
              </div>
            </div>
          </motion.div>
        ))}
      </div>
    </div>
  );
}

#### QAnything智能问答组件
```tsx
// components/AIAssistant.tsx
import { useState, useRef, useEffect } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { sendMessageToQAnything } from '@/lib/api/qanything';

export default function AIAssistant() {
  const [messages, setMessages] = useState([
    {
      role: 'assistant',
      content: '你好！我是你的编程助手，有什么可以帮助你的吗？',
      timestamp: new Date().toISOString()
    }
  ]);
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const messagesEndRef = useRef(null);
  
  // 自动滚动到最新消息
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);
  
  const handleSendMessage = async () => {
    if (!input.trim() || isLoading) return;
    
    const userMessage = {
      role: 'user',
      content: input,
      timestamp: new Date().toISOString()
    };
    
    setMessages(prev => [...prev, userMessage]);
    setInput('');
    setIsLoading(true);
    
    try {
      // 调用QAnything API获取响应
      const response = await sendMessageToQAnything(input);
      
      const assistantMessage = {
        role: 'assistant',
        content: response.answer,
        timestamp: new Date().toISOString(),
        sources: response.sources // 可选：显示答案来源
      };
      
      setMessages(prev => [...prev, assistantMessage]);
    } catch (error) {
      const errorMessage = {
        role: 'assistant',
        content: '抱歉，我暂时无法回答你的问题。请稍后再试。',
        timestamp: new Date().toISOString()
      };
      
      setMessages(prev => [...prev, errorMessage]);
    } finally {
      setIsLoading(false);
    }
  };
  
  return (
    <div className="ai-assistant">
      <div className="assistant-header">
        <h3>编程智能助手</h3>
        <p className="assistant-subtitle">基于QAnything的知识库问答</p>
      </div>
      
      <div className="messages-container">
        {messages.map((message, index) => (
          <AnimatePresence key={index}>
            <motion.div
              initial={{ opacity: 0, y: 10 }}
              animate={{ opacity: 1, y: 0 }}
              exit={{ opacity: 0, y: -10 }}
              className={`message ${message.role}`}
            >
              <div className="message-avatar">
                {message.role === 'user' ? '👤' : '🤖'}
              </div>
              <div className="message-content">
                <p>{message.content}</p>
                {message.sources && message.sources.length > 0 && (
                  <div className="message-sources">
                    <p className="sources-label">参考来源:</p>
                    <ul>
                      {message.sources.map((source, i) => (
                        <li key={i}>
                          <a href={source.url} target="_blank" rel="noopener noreferrer">
                            {source.title}
                          </a>
                        </li>
                      ))}
                    </ul>
                  </div>
                )}
                <div className="message-time">
                  {new Date(message.timestamp).toLocaleTimeString()}
                </div>
              </div>
            </motion.div>
          </AnimatePresence>
        ))}
        <div ref={messagesEndRef} />
        
        {isLoading && (
          <div className="typing-indicator">
            <div className="dot"></div>
            <div className="dot"></div>
            <div className="dot"></div>
          </div>
        )}
      </div>
      
      <div className="input-container">
        <input
          type="text"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyPress={(e) => e.key === 'Enter' && handleSendMessage()}
          placeholder="输入你的问题..."
          disabled={isLoading}
        />
        <button onClick={handleSendMessage} disabled={isLoading}>
          {isLoading ? '发送中...' : '发送'}
        </button>
      </div>
    </div>
  );
}
